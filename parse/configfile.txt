server {

    listen 8081
    listen 8082

    server_name example.com
    host localhost
    
    root /home/vscode/web-server/

    autoindex on
    index index.html

    # Configure error pages
    error_page 404 /404.html

    location /image/ {
        alias /../web-server/

        methods GET
        autoindex on
        cgi_extention   .py
        cgi_bin      dd

    }

        methods GET POST DELETE

        root slm2
        index slm3
        methods POST
        autoindex off
        cgi_extention   .dd
        cgi_bin      dff   
        return 305 /home/vscode/web-server
    }
}
